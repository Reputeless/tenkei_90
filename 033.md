# 033 - Not Too Bright (★2)

```cpp
#include <iostream>

int main()
{
	// 縦 H, 横 W
	int H, W;
	std::cin >> H >> W;

	// 解答の出力
	if ((H == 1) || (W == 1)) // 2x2 領域が存在しないコーナーケース
	{
		// 全部点灯してよい
		std::cout << (H * W) << '\n';
	}
	else
	{
		// *.*.*.*  :: (7 + 1) / 2 = 4
		// .......
		// *.*.*.*
		// .......
		// *.*.*.*
		// .......
		// ::
		// (6 + 1) / 2 = 3

		// 点灯数 = 横方向の点灯数 * 縦方向の点灯数
		const int a = ((W + 1) / 2) * ((H + 1) / 2);

		std::cout << a << '\n';
	}
}
```
