# 007 - CP Classes (★3)

## 解答

```cpp
#include <iostream>
#include <vector>
#include <algorithm> // std::distance(), std::lower_bound()
 
int main()
{
	// N 個のクラス
	int N;
	std::cin >> N;
 
	// 対象レーティング
	std::vector<int> A(N);
	for (auto& a : A)
	{
		std::cin >> a;
	}
	// 小さい順にソート
	std::sort(A.begin(), A.end());
 
	// Q 人の入塾
	int Q;
	std::cin >> Q;
 
	for (int i = 0; i < Q; ++i)
	{
		// 入塾者のレーティング
		int b;
		std::cin >> b;
 
		const int lower = static_cast<int>(std::distance(A.begin(), std::lower_bound(A.begin(), A.end(), b)));
 
		// 候補のクラス 1
		const int c1 = std::max(0, lower - 1);
 
		// 候補のクラス 2
		const int c2 = std::min(lower, static_cast<int>(A.size() - 1));
 
		// 候補のクラス 1 のコスト
		const int c1Cost = std::abs(A[c1] - b);
 
		// 候補のクラス 2 のコスト
		const int c2Cost = std::abs(A[c2] - b);
 
		// 小さいほうを解答として出力
		std::cout << std::min(c1Cost, c2Cost) << '\n';
	}
}
```
